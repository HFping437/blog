            timer = setTimeout(() => {
                fn.apply(self, arguments);
            }, delay);
        };
    },

                                        FNThrottle: function(fn, delay) {
        let valid = true;

        return function() {
            if (!valid) {
                return false;
            }

            let self = this;
            valid = false;
            setTimeout(() => {
                fn.apply(self, arguments);
                valid = true;
            }, delay);
        };
    },

                                            //base64转文件流
    dataURLtoBlob (dataurl) {
      let bstr = window.atob(dataurl);
      let n = bstr.length;
      let u8arr = new Uint8Array(n);
      while (n--) {
        u8arr[n] = bstr.charCodeAt(n);
      }
      return new Blob([u8arr]);
    },

                                        # ?.可选链运算符和??空值合并运算符??=空值赋值运算符

                                        # 可选链运算符（?.）

                                        实际开发中，我们想要获取某个对象的属性，为了防止报错，需要先判断一下对象是否存在，如下：

                                        let a
let b
if(!!a) {
	b = a.name
}else {
	b = undefined
}

                                        如上判断比较麻烦，可以使用?.运算符来简化。如下：

                                        let a
let b = a?.name

                                        上面表达式的意思为：只有当a存在，并且a的name属性有值的时候，才会将值赋值给b，否则将undefined赋值给b，重要的是，不管a存不存在，都不会报错。

                                        我们还可以这样：

                                        let a;
let b;
b = a?.name?.age?.haha?.就是不报错
a?.b?.c(“还是不报错”)

                                        我们看到,你后面可以无限接龙下去,不论有多少属性,只要有最后可以访问到属性,访问到最终的结果,就会赋值给b,否则,就把undefined赋值给b.(这样才有资格叫链式结构嘛)。

                                        # 空值合并运算符（??）

                                        读取对象属性的时候，如果某个属性的值是 null或 undefined，有时候需要为它们指定默认值。常见做法是通过 ||运算符指定默认值。

                                        let a = 0
let b = a||'default'

                                        我们的原意是，只要属性的值为 null或 undefined，默认值就会生效，但是属性的值如果为空字符串或 false或 0，默认值也会生效。

                                        我们可以使用??运算符解决：

                                        let a = 0
let b = a??'default'

                                        上面的表达式意思为：当a的值为null或undefined以外的任何值时b都会赋值为a，否则赋值default。

                                        # 空值赋值运算符（??=）

                                        let b = '你好';
let a = 0
let c = null;
let d = ’123‘
b ??= a;  // b = “你好”
c ??= d  // c = '123'

                                        ??=表达式的意思为：当且仅当表达式左侧的值为null或undefined时，将表达式右侧赋值给左侧。

                                        dynamicLoad(url, callback) {
    let oScript = document.createElement('script');
    oScript.src = url;
    let ohead = document.querySelector('head');
    ohead.appendChild(oScript);
    oScript.onload = () => {
        callback && callback();
    }
}

                                        # 对象深拷贝

                                         _deepCloneConfig(obj) {
     //判断拷贝的要进行深拷贝的是数组还是对象，是数组的话进行数组拷贝，对象的话进行对象拷贝
     var objClone = Array.isArray(obj) ? [] : {};
     //进行深拷贝的不能为空，并且是对象或者是
     if (obj && typeof obj === "object") {
         for ( let key in obj) {
             if (obj.hasOwnProperty(key)) {
                 if (obj[key] && typeof obj[key] === "object") {
                     objClone[key] = this._deepCloneConfig(obj[key]);
                 } else {
                     objClone[key] = obj[key];
                 }
             }
         }
     }
     return objClone;
 }

                                        # 字符串转对象

                                        JSON.parse() 是 JavaScript 中的一个内置函数，用于将 JSON 字符串转换为 JavaScript 对象。JSON.parse() 方法解析一个 JSON 字符串，可选地转换生成的值及其属性，并返回生成的值。

                                        const json = '{"result":true, "count":42}';
const obj = JSON.parse(json);

console.log(obj.count);
// expected output: 42

console.log(obj.result);
// expected output: true

                                        注意：使用JSON.parse()、JSON.stringify() 需要用try...catch()捕获异常
